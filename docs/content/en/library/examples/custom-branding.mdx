### Component Theming
// components/ui/Button.tsx

<Code language="typescript" title="TypeScript">
{`interface ButtonProps {
  variant?: 'primary' | 'secondary' | 'brand'
  children: React.ReactNode
}

export const Button: FC<ButtonProps> = ({ 
  variant = 'primary', 
  children,
  ...props 
}) => {
  const variants = {
    primary: 'bg-gradient-to-r from-primary-500 to-primary-600',
    secondary: 'bg-gradient-to-r from-secondary-500 to-secondary-600',
    brand: 'bg-gradient-brand'  // Your custom brand gradient
  }

  return (
    <button 
      className={`
        px-6 py-3 rounded-[var(--border-radius-brand)] 
        font-medium text-white transition-all
        hover:scale-105 hover:shadow-lg
        ${variants[variant]}
      `}
      {...props}
    >
      {children}
    </button>
  )
}`}
</Code>

### Logo Variants
// components/brand/Logo.tsx

<Code language="typescript" title="TypeScript">
{`interface LogoProps {
  variant?: 'full' | 'icon' | 'text'
  size?: 'sm' | 'md' | 'lg'
  className?: string
}

export const Logo: FC<LogoProps> = ({ 
  variant = 'full', 
  size = 'md',
  className = ''
}) => {
  const sizes = {
    sm: 'h-8',
    md: 'h-12',
    lg: 'h-16'
  }

  if (variant === 'icon') {
    return (
      <img 
        src="/brand/logo-icon.svg" 
        alt="Your Company"
        className={`${sizes[size]} w-auto ${className}`}
      />
    )
  }

  if (variant === 'text') {
    return (
      <span className={`font-brand font-bold text-brand ${className}`}>
        Your Company
      </span>
    )
  }

  return (
    <div className={`flex items-center ${className}`}>
      <img 
        src="/brand/logo-icon.svg" 
        alt=""
        className={`${sizes[size]} w-auto mr-3`}
      />
      <span className={`font-brand font-bold text-brand`}>
        Your Company
      </span>
    </div>
  )
}`}
</Code>

## üì± Multi-Brand Configuration

For managing multiple brands:

// lib/brands.ts

<Code language="typescript" title="TypeScript">
{`export const brands = {
  default: {
    name: 'Ring Platform',
    colors: { primary: '#3b82f6' },
    logo: '/logos/ring.svg'
  },
  client1: {
    name: 'Client Company',
    colors: { primary: '#10b981' },
    logo: '/logos/client1.svg'
  }
}

export function getBrand(hostname: string) {
  const brandMap = {
    'client1.yourplatform.com': 'client1',
    'demo.yourplatform.com': 'demo'
  }
  
  return brands[brandMap[hostname]] || brands.default
}`}
</Code>
